```go
import (
    "strings"
    "strconv"
    "sort"
)

func solution(s string) string {
    sp := strings.Split(s, " ")
    
    items := make([]int, len(sp))
    
    for i, v := range sp {
        n, _ := strconv.Atoi(v)
        items[i] = n
    }
    
    sort.Slice(items, func(i,j int) bool {
        return items[i] < items[j]
    })
    
    retString := strconv.Itoa(items[0]) + " " + strconv.Itoa(items[len(items)-1])
    return retString
}
```
