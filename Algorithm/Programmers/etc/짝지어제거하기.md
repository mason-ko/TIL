```javascript
function solution(s)
{
    if (s.length %2 === 1) {
        return 0;
    }
    
    let ln = s.length;
    var arr = [];
    
    for (let i=0;i<ln;i++) {
        let now = s[i];
        
        if (arr.length === 0) {
            arr.push(now);
            continue;
        }
        
        let old = arr.pop();        
        
        if (old !== now) {
            arr.push(old);
            arr.push(now);   
        }
    }
    
    return arr.length === 0 ? 1 : 0;
}
```
