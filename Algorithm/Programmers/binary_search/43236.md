import (
    "sort"
)

func solution(distance int, rocks []int, n int) int {
    sort.Ints(rocks)
    dis := make([]int, len(rocks)+1)
    for i, v := range rocks {
        if i == 0 {
            dis[i] = v
        } else {
            dis[i] = v - rocks[i-1]
        }
    }
    dis[len(rocks)] = distance - rocks[len(rocks)-1]

    start := 1
    end := distance
    ret := 1
    for start <= end {
        mid := (start+end)/2
        copyDis := make([]int, len(rocks)+1)
        copy(copyDis, dis)
        bc := 0
        for i := 0; i<len(copyDis)-1; i++ {
            r := copyDis[i]
            if mid > r {
                copyDis[i+1] = copyDis[i+1] + r
                copyDis[i] = 0
                bc++
                if bc > n {
                    break
                }
            }
        }
        if bc <= n {
            if ret < mid {
                ret = mid
            }
            start = mid + 1
        } else if bc > n {
            end = mid - 1
        }
    }
    return ret
}
