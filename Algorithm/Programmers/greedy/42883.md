```go
func solution(number string, k int) string {
	var arr []byte
    nl := len(number)
    ln := nl - k
	for i := 0; i < nl; i++ {
		n := number[i]
		for {
            aln := len(arr)
			if aln == 0 || arr[aln-1] >= n || k == 0 {
				arr = append(arr, n)
				break
			} else {
				k--
				arr = arr[:aln-1]
			}
		}
	}
    return string(arr)[:ln]
}
```
